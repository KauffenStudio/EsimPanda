   export interface ESIMProvider {
     listDestinations(): Promise<NormalizedDestination[]>;
     listPackages(destinationIso: string): Promise<NormalizedPackage[]>;
     purchase(packageId: string, quantity: number): Promise<NormalizedPurchase>;
     getStatus(iccid: string): Promise<NormalizedPurchase>;
     topUp(iccid: string, packageId: string): Promise<NormalizedPurchase>;
   }

   export function createProvider(): ESIMProvider {
     return new CelitechAdapter();
   }

3. Create `src/lib/esim/celitech-adapter.ts`:
   - Import Celitech from 'celitech-sdk'
   - Class CelitechAdapter implements ESIMProvider
   - Constructor creates Celitech client with process.env.CELITECH_CLIENT_ID and process.env.CELITECH_CLIENT_SECRET
   - listDestinations(): calls this.client.destinations.list(), maps each destination to { name: d.name, iso: d.isoCode, region: d.region ?? 'unknown' }
   - listPackages(destinationIso): calls this.client.packages.list({ destination: destinationIso }), maps to NormalizedPackage with wholesalePriceCents as Math.round(price * 100)
   - purchase(packageId, quantity): calls this.client.purchases.create(), maps response to NormalizedPurchase
   - getStatus(iccid): calls this.client.esim.getEsim({ iccid }), maps to NormalizedPurchase
   - topUp(iccid, packageId): calls this.client.purchases.topUp(), maps to NormalizedPurchase
   - All methods normalize provider-specific types to Normalized* types -- NO celitech-sdk types leak out
